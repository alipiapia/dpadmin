
【用户/会员】模块【周期：一周（0626-0630）】
	1，用户基础信息管理；
		a，建立用户/会员表@dp_user
			CREATE TABLE `dp_user` (
			`id` int(11) unsigned NOT NULL AUTO_INCREMENT,
			`username` varchar(32) NOT NULL COMMENT '用户名',
			`nickname` varchar(32) NOT NULL DEFAULT '' COMMENT '昵称',
			`password` varchar(96) NOT NULL DEFAULT '' COMMENT '登录密码',
			`paypass` varchar(96) NOT NULL DEFAULT '' COMMENT '支付密码',
			`email` varchar(64) NOT NULL DEFAULT '' COMMENT '邮箱地址',
			`mobile` varchar(11) NOT NULL DEFAULT '' COMMENT '手机号码',
			`ref_mobile` varchar(11) NOT NULL DEFAULT '' COMMENT '推荐人手机号码',
			`balance` decimal(11,2) unsigned NOT NULL DEFAULT '0.00' COMMENT '余额',
			`score` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '积分',
			`signup_ip` bigint(20) unsigned NOT NULL DEFAULT '0' COMMENT '注册ip',
			`create_time` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '注册时间',
			`update_time` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '更新时间',
			`last_login_time` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '最后一次登录时间',
			`last_login_ip` bigint(20) unsigned NOT NULL DEFAULT '0' COMMENT '登录ip',
			`status` tinyint(2) NOT NULL DEFAULT '1' COMMENT 状态：0禁用，1启用,
			PRIMARY KEY (`id`)
			) ENGINE=MyISAM AUTO_INCREMENT=1 DEFAULT CHARSET=utf8 COMMENT='用户/会员表';
		b,用户注册;【完の结】
		c，用户登录;【完の结】
		d,用户资料修改（昵称、邮箱等）；
		e,登录密码、支付密码修改；
		f,收货地址；
			建立收货地址表@dp_user_address
			CREATE TABLE `dp_user_address` (
			`id` int(11) unsigned NOT NULL AUTO_INCREMENT,
			`uid` int(11) unsigned NOT NULL DEFAULT '' COMMENT '用户id',
			`name` varchar(1000) NOT NULL DEFAULT '' COMMENT '地址',
			`postcode` varchar(10) NOT NULL DEFAULT '' COMMENT '邮编',
			`is_defaut` tinyint(1) NOT NULL DEFAULT '0' COMMENT '是否为默认地址：0否 1是',
			`create_time` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '添加时间',
			`update_time` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '更新时间',
			`status` tinyint(2) NOT NULL DEFAULT '1' COMMENT 状态：0禁用，1启用,
			PRIMARY KEY (`id`)
			) ENGINE=MyISAM AUTO_INCREMENT=1 DEFAULT CHARSET=utf8 COMMENT='用户收货地址表';
		
	2,账户变动；
		a,建立用户资金变动表@dp_user_account;
			CREATE TABLE `dp_user_account` (
			`id` int(11) unsigned NOT NULL AUTO_INCREMENT,
			`put_id` int(11) unsigned NOT NULL DEFAULT '' COMMENT '发起方用户id',
			`get_id` int(11) unsigned NOT NULL DEFAULT '' COMMENT '接收方用户id',
			`count` decimal(11,2) unsigned NOT NULL DEFAULT '0.00' COMMENT '变动余额',
			`type` tinyint(1) NOT NULL DEFAULT '0' COMMENT '变动类型：0充值 1返利 2提成',
			`title` varchar(255) NOT NULL DEFAULT '' COMMENT '变动说明',
			`create_time` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '创建时间',
			`update_time` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '更新时间',
			PRIMARY KEY (`id`)
			) ENGINE=MyISAM AUTO_INCREMENT=1 DEFAULT CHARSET=utf8 COMMENT='用户资金变动表';
